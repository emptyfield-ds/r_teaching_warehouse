Working with descriptive tables

descriptives <- diabetes |>
  filter(!is.na(glyhb)) |>
  mutate(
    diabetic = case_when(
      glyhb >= 6.5 ~ "Diabetic", 
      glyhb < 6.5 ~ "Healthy",
      NA ~ NA_character_
    ),
    bmi = (weight / height^2) * 703
  ) |>
  group_by(diabetic) |> 
  summarise(
    n = n(), 
    across(
      c(glyhb, bmi, age), 
      \(.x) mean(.x, na.rm = TRUE)
    ))

Argument Description
